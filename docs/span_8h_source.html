438/* Comparison operators */
439// Implementation note: the implementations of == and < are equivalent to
440// 4-legged std::equal and std::lexicographical_compare respectively
441
442template <typename T, std::size_t X, typename U, std::size_t Y>
443constexpr bool operator==(span<T, X> lhs, span<U, Y> rhs)
444{
445 if (lhs.size() != rhs.size()) {
446 return false;
447 }
448
449 for (std::size_t i = 0; i < lhs.size(); i++) {
450 if (lhs[i] != rhs[i]) {
451 return false;
452 }
453 }
454
455 return true;
456}
457
458template <typename T, std::size_t X, typename U, std::size_t Y>
459constexpr bool operator!=(span<T, X> lhs, span<U, Y> rhs)
460{
461 return !(lhs == rhs);
462}
463
464template <typename T, std::size_t X, typename U, std::size_t Y>
465constexpr bool operator<(span<T, X> lhs, span<U, Y> rhs)
466{
467 // No std::min to avoid dragging in <algorithm>
468 const std::size_t size = lhs.size() < rhs.size() ? lhs.size() : rhs.size();
469
470 for (std::size_t i = 0; i < size; i++) {
471 if (lhs[i] < rhs[i]) {
472 return true;
473 }
474 if (lhs[i] > rhs[i]) {
475 return false;
476 }
477 }
478 return lhs.size() < rhs.size();
479}
