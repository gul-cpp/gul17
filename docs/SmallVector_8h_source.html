1076 void reserve(SizeType new_capacity)
1077 {
1078 if (new_capacity <= capacity_)
1079 return;
1080
1081 // Allocate aligned memory for the new "outer" capacity.
1082 auto* new_data = allocate_space_for_elements(new_capacity);
1083 auto _ = finally([&new_data]() { deallocate_space_for_elements(new_data); });
1084
1085 auto* d_end = data_end();

1089 if (is_storage_allocated())

1091
1092 data_ptr_ = new_data;
1093 new_data = nullptr;
1094 capacity_ = new_capacity;

1421 {
1422 const auto remaining_space = std::numeric_limits<SizeType>::max() - capacity_;
1423

1438 Iterator insert_single_value(ConstIterator pos, T&& value)
1439 {
1440 if (pos == end())
1441 {
1442 push_back(std::forward<T>(value));

1482 SizeType make_space_at_idx_for_n_elements(SizeType idx, SizeType num_elements)
1483 {
1484 const auto new_size = size_ + num_elements;
1485
1486 if (new_size > capacity_)
1487 reserve(new_size);
1488
1489 auto data_ptr = data();
1490
1491 // We have a total number of elements that need to be shifted backwards,
1492 // of which some need to be move-initialized at the end of the vector
1493 // and some simply need to be moved from one element to another.
1494 const SizeType num_shifted = size_ - idx;
1495 const SizeType num_move_initialized = std::min(num_shifted, num_elements);
1496 const SizeType num_moved = num_shifted - num_move_initialized;
1497
1498 // Move-initialize elements in uninitialized space at the back of the container
1499 const auto from_ptr = data_ptr + size_ - num_move_initialized;
1500 const auto to_ptr = from_ptr + num_elements;
1501 for (SizeType i = 0; i != num_move_initialized; ++i)
1502 ::new(static_cast<void*>(to_ptr + i)) ValueType(std::move(*(from_ptr + i)));
1503
1504 // Move elements backwards within old vector size
1505 std::move_backward(data_ptr + idx, data_ptr + idx + num_moved, to_ptr);
1506
1507 return num_move_initialized;
1508 }
